
 class Solution {
    public double myPow(double x, int n) {
        long N = n;  // convert to long to avoid overflow when n = -2^31
        
        if (N < 0) {
            x = 1 / x;
            N = -N;
        }
        
        double ans = 1;
        
        while (N > 0) {
            if ((N & 1) == 1) {  
                ans *= x;      
            }
            x *= x;           
            N >>= 1;          
        }
        
        return ans;
    }
}
